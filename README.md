      private static void bubbleSort(int[] arr) {
          int n = arr.length;
          for(int i = 0; i < n; i++){
              for (int j = 0; j < n; j++){
                  if(arr[j] > arr[j+1]){
                      int tmp = arr[j];
                      arr[j] = arr[j+1];
                      arr[j+1] = tmp;
                  }
              }
          }
